  <name>Task 1: Phát triển Security Gate Script</name>
  <files>[java-repo]/scripts/security-gate-check.sh</files>
  <action>
    Tạo `scripts/security-gate-check.sh` trong Java repo:
    ```bash
    #!/bin/bash
    # Security Gate: Fail nếu có HIGH/CRITICAL findings trong bất kỳ SARIF file nào.
    set -euo pipefail

    FAIL_BUILD=0
    CRITICAL_COUNT=0
    HIGH_COUNT=0
    SARIF_FILES=$(find . -name "*.sarif" -not -path "*/node_modules/*")

    if [ -z "$SARIF_FILES" ]; then
      echo "⚠️  Không tìm thấy SARIF files."
      exit 0
    fi

    for FILE in $SARIF_FILES; do
      echo "Evaluating $FILE..."

      # Đếm findings level "error" (CRITICAL/HIGH) hoặc security-severity >= 7.0
      COUNT=$(jq '[.runs[].results[] |
        select(.level == "error" or
               ((.properties."security-severity" // "0") | tonumber) >= 7.0)
      ] | length' "$FILE" 2>/dev/null || echo "0")

      if [ "$COUNT" -gt "0" ]; then
        echo "::error file=$FILE::Found $COUNT High/Critical findings."
        FAIL_BUILD=1
        HIGH_COUNT=$((HIGH_COUNT + COUNT))
      fi
    done

    echo "GATE_RESULT=$([ $FAIL_BUILD -eq 0 ] && echo 'PASSED' || echo 'FAILED')" >> $GITHUB_ENV
    echo "HIGH_CRITICAL_COUNT=$HIGH_COUNT" >> $GITHUB_ENV

    exit $FAIL_BUILD
    ```
    - Chmod executable: `chmod +x scripts/security-gate-check.sh`
  </action>
  <verify>
    <automated>bash scripts/security-gate-check.sh</automated>
  </verify>
  <done>Script chạy được, exit 1 đúng khi có High/Critical, exit 0 khi clean.</done>

<task type="auto">
  <name>Task 2: Tích hợp Security Gate và PR Comment vào Workflow</name>
  <files>[java-repo]/.github/workflows/security-scans.yml</files>
  <action>
    Thêm job `security-gate` vào cuối workflow:
    ```yaml
    security-gate:
      runs-on: ubuntu-latest
      needs: [semgrep, codeql, eslint, spotbugs, dependency-check]
      if: always()
      permissions:
        pull-requests: write
        actions: read
      steps:
        - uses: actions/checkout@v4

        - name: Download All SARIF Artifacts
          uses: actions/download-artifact@v4
          with:
            pattern: security-results-*
            merge-multiple: true

        - name: Run Security Gate
          run: bash scripts/security-gate-check.sh

        - name: Post PR Comment
          if: github.event_name == 'pull_request'
          env:
            GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}
          run: |
            STATUS="${GATE_RESULT:-UNKNOWN}"
            COUNT="${HIGH_CRITICAL_COUNT:-0}"
            ICON=$([ "$STATUS" = "PASSED" ] && echo "✅" || echo "❌")

            gh pr comment ${{ github.event.pull_request.number }} --body "
            ## $ICON Security Gate: $STATUS

            | Metric | Value |
            |--------|-------|
            | High/Critical Findings | $COUNT |
            | Full Results | [Security Tab](${{ github.server_url }}/${{ github.repository }}/security/code-scanning) |

            > Kết quả chi tiết đã được gửi về MCP Gateway để phân tích AI.
            "
    ```
  </action>
  <verify>
    <automated>gh workflow view security-scans.yml | grep "security-gate"</automated>
  </verify>
  <done>Security gate job chạy sau tất cả scan jobs, post PR comment.</done>
