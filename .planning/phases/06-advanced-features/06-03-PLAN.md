      const before = await page.locator('[data-testid="finding-row"]').count();

      await page.request.post('http://localhost:8001/test/inject-finding', {
        data: { severity: 'HIGH', rule_id: 'test-sqli', message: 'SQL Injection detected' }

      // Đợi polling cycle (timeout 20s)
      await page.waitForFunction(
        (n) => document.querySelectorAll('[data-testid="finding-row"]').length > n,

      );

      // Warning toast phải xuất hiện

    });
