    Tạo `ApprovalDialog.tsx` dùng Shadcn Dialog:
    ```tsx
    interface ActionDialogProps {
      findingId: string
      open: boolean
      onClose: () => void
      onConfirm: (justification: string) => Promise<void>
      title: string
      confirmLabel: string
      confirmVariant: 'default' | 'destructive'
    }

    // ApprovalDialog.tsx
    export function ApprovalDialog({ findingId, open, onClose }) {
      const [justification, setJustification] = useState('');
      const isValid = justification.trim().length >= 20;

      const handleConfirm = async () => {
        await submitCommand({ command: '/approve', finding_id: findingId, justification });
        notify.commandSuccess('Finding đã được phê duyệt');
        onClose();
      };

      return (
        <ActionDialog
          title="Phê duyệt Finding"
          confirmLabel="Xác nhận phê duyệt"
          confirmVariant="default"
          ...
        >
          <Textarea
            placeholder="Lý do phê duyệt (tối thiểu 20 ký tự)..."
            value={justification}
            onChange={e => setJustification(e.target.value)}
          />
          {!isValid && justification.length > 0 && (
            <p className="text-red-500 text-sm">Cần ít nhất 20 ký tự</p>
          )}
        </ActionDialog>
      );
    }
    ```

    - Title: "Thu hồi phê duyệt"
